Indica la diferencia entre los siguientes métodos, y demuestra su uso con algunos arrays: Array.prototype.forEach(), Array.prototype.every(), Array.prototype.some() y Array.prototype.filter().

Array.prototype.forEach()

Esta función ejecuta la función que se le indica una vez por cada elemento que tenga el array.

Array.prototype.every()

Esta función devuelve un booleano si todos los elementos del array cumplen una condición implementada por la función dada.

Array.prototype.some()

Parecido al anterior pero devuelve True en caso de que al menos uno cumpla la condición.

Array.prototype.filter()

Crea un nuevo array con los elementos que cumplan la condición implementada por la condición dada. 
Para ver la salida de las funciones abre la consola.

----

let ejemploArray1 = ["hola",1,2,3,NaN,"adios",undefined];
ejemploArray1.forEach(muestraValor);

let ejemploArray2 = [4,6,7,8,9,10];
console.log("Every: "+ejemploArray2.every(masTres));
console.log("Some: "+ejemploArray2.some(masSeis));
console.log("Filter: "+ejemploArray2.filter(menosOcho));


function muestraValor(valor){
    console.log(valor);
}

function masTres(valor){
    return valor > 3;
}

function masSeis(valor){
    return valor > 6;
}

function menosOcho(valor){
    return valor < 8;
}
